rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /opiniones/{userId}/{allPaths=**} {
      allow write: if request.auth != null && request.auth.uid == userId
                   && request.resource.size < 10 * 1024 * 1024
                   && request.resource.contentType == 'application/pdf';
      allow read: if false;
    }
  }
}
